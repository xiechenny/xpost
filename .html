if
if test -z $arg : test if none

file
${file%.*} : get file name without extension
${file##*.} : get extension of file
${filepath%/*} : get directory path of this file
${filepath##*/} : get full filename of filepath

string
${#string} : string length
${string:position}
${string:position:length} : cut string from position, set length if need
${arg/substr/replacestr} : replace first substr; ${arg//...}, - replace all
${arg#substr} : cut, # from start, % from end.
repeat once # - cut shortest matching, repeat twice ## - cut longest
